        def number_of_instructions
        
                if @inst_num != -1
                        return @inst_num
                end
        
                start_offset = address
                end_offset = last_offset.address
                ctr = 0
                
                for addr in (start_offset ... end_offset)
                        
                        flags = @ida.getFlags(addr)
                        
                        if @ida.isHead(flags) and @ida.isCode(flags)
                                ctr = ctr + 1
                        end
                        
                end
                
                @inst_num = ctr
                
                return ctr
        end
