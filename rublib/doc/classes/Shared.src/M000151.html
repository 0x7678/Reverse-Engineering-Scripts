        def crefs_from
                refs = []
                
                curr = @ida.get_first_cref_from(address)
                
                while curr != @ida.BADADDR
                        refs << Instruction.new(@ida, curr)
                        curr = @ida.get_next_cref_from(address, curr)
                end
                
                refs
        end
