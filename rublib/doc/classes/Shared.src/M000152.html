        def drefs_to
                refs = []
                
                curr = @ida.get_first_dref_to(address)
                
                while curr != @ida.BADADDR
                        refs << Offset.new(@ida, curr)
                        curr = @ida.get_next_dref_to(address, curr)
                end
                
                refs
        end
