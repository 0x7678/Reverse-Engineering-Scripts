        def mnemonic(instruction)
                if @ida.ua_mnem(instruction.offset) != nil:
                        mnem = @ida.cmd.get_canon_mnem

                        byte = instruction.byte
                        
                        if byte == 0xF3
                                if mnem == "scas"
                                        return "repe " + mnem
                                else
                                        return "rep " + mnem
                                end
                        elsif byte == 0xF2
                                        return "repne " + mnem
                        elsif byte == 0xF0
                                        return "lock " + mnem
                        else
                                return mnem
                        end
                else
                        return nil
                end
        end
